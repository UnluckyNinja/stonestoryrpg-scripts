var lc = import MyScripts/locheck
var rsc = import MyScripts/resources
var eq = import MyScripts/equips
var pot = import MyScripts/potions

?lc.isTemple()
  pot.usehealthPotion()
  eq.equipWalk()
  ?loc.stars <= 5
    equipL triskelion
  rsc.pickups()
  rsc.harvests()

  ?foe.distance < 30
    ?foe = boss
      equipBoss()
    :
      ?loc.stars <= 5
        eq.equipSpeed()
      :
        eq.equipGeneral()

  ?foe = boss
    ?foe.state = 33
      ?loc.stars > 5
        ?foe.time >= 15
          equipR mind stone
      :
        ?foe.time >= 15
          equipR mind stone
    :?foe.state = 112
      ?foe.time = 80
        equipR mind stone
    :?foe.state = 4
      eq.equipFarm()

func equipBoss()
  ?eq.isUsingSkill()
    return

  equipL ice sword dI
  ?hp < maxhp
    equipR vigor sword dL
  :
    equipR ice sword D

  ?foe.distance < 9
    ?eq.tryAblities()
      return
  :
    ?foe.state ! 33
      eq.equipSpeed()
