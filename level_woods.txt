func equipBoss()
  ?eq.isUsingSkill()
    return
  ?foe = phase1
    ?foe = boss & (foe.distance < 9) & eq.tryAblities()
      return
    ?foe = boss
      ?foe.distance > 5 // get close
        equipR dashing shield
        equipL sword
        return
    eq.equipDebuff()
    eq.equipLeech()
  :
    ?foe.distance > 5 // get close
      equipR dashing shield
      equipL sword
      return
    ?foe.buffs.string = "mirror"
      equipR compound shield
      equipL vigor long sword dL
    :
      ?foe = boss & (foe.distance < 9) & eq.tryAblities()
        return
      ?foe.state = 32 & foe.debuffs.string ! debuff_damage
        equipR sword dP
        equipL sword dL
      :
        equipR sword dI
        equipL sword dL
