func equipBoss()
  ?eq.isUsingSkill()
    return
  ?foe = boss & (foe.distance < 9) & eq.tryAblities()
    return
  ?loc.stars < 6
    ?foe = boss
      ?foe.distance > 5
        equipR dashing shield
        equipL sword
        return
      ?foe.distance > 2
        equip shovel
        return
    :
      equipR aether long sword dU
      equipL aether -long sword dU
      return
    var maxLayer = 1
    ?foe = boss
      maxLayer = 1
    ?foe.debuffs.string ! dot
      equip fire socketed crossbow dF
    :
      ?hp < maxhp
        equipL vigor sword
        equipR vigor wand
      :
        equipL fire sword -long D
        ?armor < 10
          ?item.right.state ! 2
            equipR compound shield
        :
          equipR fire sword -long dF
  :?loc.stars < 11
    ?foe = boss
      ?foe.distance > 5
        equipR dashing shield
        equipL sword
        return
      ?hp < maxhp
        ?debuffs.string = chill
          equipR shield ah
        :
          equipR sword dL +0
      :
        eq.equipDebuff()
      eq.equipLeech()
    :
      ?foe.count > 2
        equip socketed crossbow aether dU
      :?foe.distance > 9
        equipR dashing shield
        equipL sword
      :
        ?foe.distance > 5
          eq.staffDash()
        ?hp < maxhp
          ?debuffs.string = chill
            equipR shield ah
          :
            equipR sword -long dL +0
          equipL sword -long dL +6
        :
          equipR aether long sword dU
          equipL aether -long sword dU
