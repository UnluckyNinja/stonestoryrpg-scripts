func equipBoss()
  ?loc.stars < 6
    ?foe = boss
      ?foe.distance > 5
        equipR dashing shield
        equipL sword
        return
      ?foe.distance > 2
        equip shovel
        return
    :
      equipR aether long sword dU
      equipL aether -long sword dU
      return
    var maxLayer = 1
    ?foe = boss
      maxLayer = 1
    ?foe.debuffs.string ! dot
      equip fire socketed crossbow dF
    :
      ?hp < maxhp
        equipL vigor sword
        equipR vigor wand
      :
        equipL fire sword -long D
        ?armor < 10
          ?item.right.state ! 2
            equipR compound shield
        :
          equipR fire sword -long dF
  :?loc.stars < 11
    ?foe = boss
      ?foe.distance > 5
        equipR dashing shield
        equipL sword
        return
    :
      equipR aether long sword dU
      equipL aether -long sword dU
      return
      // ?true
    ?foe.debuffs.string ! dot
      equip fire socketed crossbow dF
    :
      equipR vigor shield ah
      ?hp < maxhp
        equipL vigor sword dL -long
      :
        equipL fire sword -long D
